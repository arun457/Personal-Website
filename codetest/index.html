//Write an interactive program that reads positive numbers until user enters “no“
                    //and then sum the numbers divisible by 4 that lie between the range of 10 and
                    //50 and finally display the count, sum & average value. [5] [065]
                    
                    #include <stdio.h>
                    #include <conio.h>
                    #include <stdlib.h>
                    #include <string.h>
                    
                    void main()
                    {
                        int n, sum = 0, count = 0;
                        float avg;
                        char no[20] = "no";
                        char choice[20];
                        system("cls");
                        printf("Enter a number or enter \"no\" to terminate the program : \n");
                        do
                        {
                            printf("Do you want to enter another number (yes/no)? : ");
                            scanf("%s", choice);
                            if (strcmp(choice, no) != 0)
                            {
                                printf("\nEnter a Number : ");
                                scanf("%d", &n);
                                if (n % 4 == 0 && n > 10 && n < 50)
                                {
                                    sum = sum + n;
                                    count++;
                                }
                            }
                    
                        } while (strcmp(choice, no) != 0);
                        printf("The number of numbers divisible by 4 is : %d\n", count);
                        printf("The sum of numbers divisible by 4 is : %d\n", sum);
                        avg = (float)sum / count;
                        printf("The average of them is : %f\n", avg);
                    }
